<script>

  // Tab switching functionality

  const tabBtns = document.querySelectorAll('.tab-btn');

  const calculators = document.querySelectorAll('.calculator');


  tabBtns.forEach(btn => {

    btn.addEventListener('click', () => {

      // Remove active class from all tabs and calculators

      tabBtns.forEach(btn => btn.classList.remove('active'));

      calculators.forEach(calc => calc.classList.remove('active'));


      // Add active class to clicked tab and corresponding calculator

      btn.classList.add('active');

      const tabName = btn.getAttribute('data-tab');

      document.getElementById(`${tabName}-calculator`).classList.add('active');

    });

  });


  // Unit toggle functionality

  const unitToggles = document.querySelectorAll('.units-toggle');


  unitToggles.forEach(toggle => {

    const buttons = toggle.querySelectorAll('button');


    buttons.forEach(btn => {

      btn.addEventListener('click', () => {

        buttons.forEach(b => b.classList.remove('active'));

        btn.classList.add('active');

        const unit = btn.getAttribute('data-unit');


        // Update units in the corresponding calculator

        const calculator = btn.closest('.calculator');

        if (unit === 'metric') {

          calculator.querySelectorAll('[id$="-weight-unit"]').forEach(el => el.textContent = 'kg');

          calculator.querySelectorAll('[id$="-height-unit"]').forEach(el => el.textContent = 'cm');

        } else {

          calculator.querySelectorAll('[id$="-weight-unit"]').forEach(el => el.textContent = 'lbs');

          calculator.querySelectorAll('[id$="-height-unit"]').forEach(el => el.textContent = 'in');

        }

      });

    });

  });


  // BMI Calculator

  document.getElementById('calculate-bmi').addEventListener('click', () => {

    const unit = document.querySelector('#bmi-calculator .units-toggle .active').getAttribute('data-unit');

    let weight = parseFloat(document.getElementById('bmi-weight').value);

    let height = parseFloat(document.getElementById('bmi-height').value);


    if (isNaN(weight) || isNaN(height)) {

      alert('Please enter valid weight and height');

      return;

    }


    // Convert to metric if needed

    if (unit === 'imperial') {

      weight = weight * 0.453592; // lbs to kg

      height = height * 2.54; // inches to cm

    }


    // Convert height from cm to m

    height = height / 100;


    const bmi = weight / (height * height);

    const roundedBMI = Math.round(bmi * 10) / 10;


    document.getElementById('bmi-value').textContent = roundedBMI;


    let category = '';

    let categoryClass = '';


    if (bmi < 18.5) {

      category = 'Underweight';

      categoryClass = 'warning';

    } else if (bmi >= 18.5 && bmi < 25) {

      category = 'Normal weight';

      categoryClass = 'success';

    } else if (bmi >= 25 && bmi < 30) {

      category = 'Overweight';

      categoryClass = 'warning';

    } else {

      category = 'Obese';

      categoryClass = 'danger';

    }


    document.getElementById('bmi-category').textContent = category;

    document.getElementById('bmi-category').className = `result-category ${categoryClass}`;


    // Update BMI marker on chart

    const marker = document.getElementById('bmi-marker');

    let position = 0;


    if (bmi < 15) position = 0;

    else if (bmi > 40) position = 100;

    else position = ((bmi - 15) / 25) * 100;


    marker.style.left = `${position}%`;


    // Highlight active category

    document.querySelectorAll('#bmi-calculator .category').forEach(el => {

      el.classList.remove('active');

      const range = el.getAttribute('data-range').split('-');

      const min = parseFloat(range[0]);

      const max = parseFloat(range[1]);


      if (bmi >= min && bmi <= max) {

        el.classList.add('active');

      }

    });


    document.getElementById('bmi-results').classList.add('active');

  });


  // Calorie Calculator - Fixed version

  let maintenanceCalories = 0; // Store the calculated value globally


  document.getElementById('calculate-calories').addEventListener('click', () => {

    const gender = document.getElementById('cal-gender').value;

    const age = parseFloat(document.getElementById('cal-age').value);

    const unit = document.querySelector('#calorie-calculator .units-toggle .active').getAttribute('data-unit');

    let weight = parseFloat(document.getElementById('cal-weight').value);

    let height = parseFloat(document.getElementById('cal-height').value);

    const activity = parseFloat(document.getElementById('cal-activity').value);


    if (isNaN(age) || isNaN(weight) || isNaN(height)) {

      alert('Please enter valid age, weight, and height');

      return;

    }


    // Convert to metric if needed

    if (unit === 'imperial') {

      weight = weight * 0.453592; // lbs to kg

      height = height * 2.54; // inches to cm

    }


    // Mifflin-St Jeor Equation

    let bmr;

    if (gender === 'male') {

      bmr = (10 * weight) + (6.25 * height) - (5 * age) + 5;

    } else {

      bmr = (10 * weight) + (6.25 * height) - (5 * age) - 161;

    }


    maintenanceCalories = Math.round(bmr * activity);


    document.getElementById('calorie-value').textContent = `${maintenanceCalories} calories/day`;

    updateAdjustedCalories(maintenanceCalories);


    document.getElementById('calorie-results').classList.add('active');

  });


  // Update adjusted calories when goal changes

  document.getElementById('cal-goal').addEventListener('change', function() {

    updateAdjustedCalories(maintenanceCalories);

  });


  function updateAdjustedCalories(maintenance) {

    const goal = document.getElementById('cal-goal').value;

    let adjusted = maintenance;

    let message = '';


    switch(goal) {

      case 'mild_loss':

        adjusted = maintenance - 250;

        message = 'Mild weight loss (0.25 kg/week)';

        break;

      case 'loss':

        adjusted = maintenance - 500;

        message = 'Weight loss (0.5 kg/week)';

        break;

      case 'extreme_loss':

        adjusted = maintenance - 1000;

        message = 'Extreme weight loss (1 kg/week)';

        break;

      case 'mild_gain':

        adjusted = maintenance + 250;

        message = 'Mild weight gain (0.25 kg/week)';

        break;

      case 'gain':

        adjusted = maintenance + 500;

        message = 'Weight gain (0.5 kg/week)';

        break;

      case 'extreme_gain':

        adjusted = maintenance + 1000;

        message = 'Extreme weight gain (1 kg/week)';

        break;

      default:

        message = 'Maintain weight';

    }


    document.getElementById('adjusted-calories').textContent = `${adjusted} calories/day (${message})`;

  }


  // 1-Rep Max Calculator

  document.getElementById('calculate-rm').addEventListener('click', () => {

    const unit = document.querySelector('#one-rep-max-calculator .units-toggle .active').getAttribute('data-unit');

    let weight = parseFloat(document.getElementById('rm-weight').value);

    const reps = parseFloat(document.getElementById('rm-reps').value);


    if (isNaN(weight) || isNaN(reps) || reps < 1 || reps > 10) {

      alert('Please enter valid weight and reps (1-10)');

      return;

    }


    // Convert to kg if needed (for calculation)

    if (unit === 'imperial') {

      weight = weight * 0.453592; // lbs to kg

    }


    // Epley Formula

    const oneRepMax = weight * (1 + (0.0333 * reps));

    const roundedMax = Math.round(oneRepMax * 10) / 10;


    // Convert back to imperial if needed

    let displayMax = roundedMax;

    let displayUnit = 'kg';


    if (unit === 'imperial') {

      displayMax = Math.round((roundedMax / 0.453592) * 10) / 10;

      displayUnit = 'lbs';

    }


    document.getElementById('rm-value').textContent = `${displayMax} ${displayUnit}`;


    // Calculate training weights

    const percentages = [0.95, 0.85, 0.75, 0.6];

    const weightElements = ['rm-95', 'rm-85', 'rm-75', 'rm-60'];


    percentages.forEach((pct, i) => {

      let trainingWeight = roundedMax * pct;


      if (unit === 'imperial') {

        trainingWeight = trainingWeight / 0.453592;

      }


      trainingWeight = Math.round(trainingWeight * 10) / 10;

      document.getElementById(weightElements[i]).textContent = `${trainingWeight} ${displayUnit}`;

    });


    document.getElementById('rm-results').classList.add('active');

  });
